판정: **예 — 지금 단계는 설계 검토가 아니라 `Vercel Route Handler + Supabase upsert` 코드 골격으로 바로 내려가면 됩니다.** 업로드 문안의 구조도 `WAHA → Vercel webhook → Supabase → Realtime board`로 이미 맞습니다. 
근거: Vercel 공식 문서는 `app/api/.../route.ts` Route Handler, `runtime/maxDuration` 설정, raw body 기반 HMAC 검증 패턴을 지원합니다. 업로드 문안의 webhook ingress 단일화 방향과도 일치합니다.
다음행동: **아래 6개 파일 + SQL만 넣으면 v1 수집/분류/카드 upsert까지 바로 시작할 수 있습니다.**

## 고정값

* **Vercel**: UI + webhook API
* **Supabase**: Postgres + Auth + Realtime
* **WAHA**: WhatsApp 입력원
* **런타임**: `nodejs`
* **v1 범위**: 수신, 저장, 분류, 카드 갱신, 보드 반영
* **v1 제외**: WhatsApp 자동 회신, 자동 reaction 발신

업로드 문안 기준으로 WAHA payload에서 `chatId`, `participant`, `replyTo`, 그룹 식별자 `@g.us`를 활용하는 방향이 핵심입니다.

## 5) `lib/waha/verify-signature.ts`

```ts
import crypto from 'crypto';

export function verifyWahaSignature(rawBody: string, headerSignature: string | null) {
  const secret = process.env.WAHA_WEBHOOK_SECRET;
  const algo = process.env.WAHA_HMAC_ALGO || 'sha256';

  if (!secret) {
    throw new Error('WAHA_WEBHOOK_SECRET is missing');
  }

  if (!headerSignature) {
    return false;
  }

  const computed = crypto.createHmac(algo, secret).update(rawBody).digest('hex');

  if (computed.length !== headerSignature.length) {
    return false;
  }

  return crypto.timingSafeEqual(
    Buffer.from(computed, 'utf8'),
    Buffer.from(headerSignature, 'utf8')
  );
}
```

Vercel 공식 문서도 webhook 검증 시 **raw body로 HMAC 계산 후 timing-safe 비교** 패턴을 사용합니다.
Sources: [Vercel request headers / webhook signature example](https://vercel.com/docs/headers/request-headers), [Vercel Functions quickstart](https://vercel.com/docs/functions/quickstart)

## 6) `lib/waha/normalize-event.ts`

```ts
export type NormalizedWahaEvent = {
  eventType: string;
  messageId: string;
  groupId: string;
  participantId: string | null;
  replyToMessageId: string | null;
  bodyRaw: string;
  sentAt: string | null;
  payload: unknown;
};

/**
 * 가정:
 * WAHA payload 안에 아래 필드 중 일부가 존재
 * - event / eventType / type
 * - data.chatId
 * - data.id
 * - data.body or data.text
 * - data.participant
 * - data.replyTo.id
 */
export function normalizeWahaEvent(payload: any): NormalizedWahaEvent | null {
  const eventType = payload?.event || payload?.eventType || payload?.type || 'unknown';
  const data = payload?.data ?? payload?.payload ?? payload;

  const groupId = data?.chatId ?? data?.chat?.id ?? data?.to ?? null;
  if (!groupId || typeof groupId !== 'string') return null;
  if (!groupId.endsWith('@g.us')) return null;

  const messageId = data?.id ?? data?.message?.id ?? null;
  if (!messageId || typeof messageId !== 'string') return null;

  const participantId =
    data?.participant ??
    data?.author ??
    data?.from ??
    null;

  const replyToMessageId =
    data?.replyTo?.id ??
    data?.replyToMessageId ??
    null;

  const bodyRaw =
    data?.body ??
    data?.text ??
    data?.caption ??
    '';

  const sentAt = normalizeTimestamp(
    data?.timestamp ??
    data?.sentAt ??
    data?.messageTimestamp ??
    null
  );

  return {
    eventType,
    messageId,
    groupId,
    participantId,
    replyToMessageId,
    bodyRaw,
    sentAt,
    payload,
  };
}

function normalizeTimestamp(value: unknown): string | null {
  if (!value) return null;

  if (typeof value === 'number') {
    const ms = value < 10_000_000_000 ? value * 1000 : value;
    return new Date(ms).toISOString();
  }

  if (typeof value === 'string') {
    const asNumber = Number(value);
    if (!Number.isNaN(asNumber)) {
      const ms = asNumber < 10_000_000_000 ? asNumber * 1000 : asNumber;
      return new Date(ms).toISOString();
    }

    const d = new Date(value);
    if (!Number.isNaN(d.getTime())) {
      return d.toISOString();
    }
  }

  return null;
}
```

이 부분은 업로드 문안의 WAHA 필드 설명을 기준으로 만든 **어댑터 계층**입니다. 실제 payload 샘플 3~5건으로 바로 미세조정하면 됩니다.

## 7) `lib/parser/classify-message.ts`

```ts
type Status = 'new' | 'assigned' | 'in_progress' | 'hold' | 'done';

export type ParsedMessage = {
  title: string;
  status: Status;
  typeCode: string;
  ownerName: string | null;
  holdReason: string | null;
  priorityScore: number;
  dedupeKey: string | null;
};

const DONE_PATTERNS = [/done/i, /completed/i, /released/i, /received/i, /collected/i];
const HOLD_PATTERNS = [/pending/i, /waiting/i, /hold/i, /blocked/i, /no reply/i, /waiting boe/i];
const PROGRESS_PATTERNS = [/ongoing/i, /under process/i, /working/i, /arranging/i, /follow ?up/i];

const TYPE_RULES: Array<{ code: string; patterns: RegExp[] }> = [
  { code: 'GATE_PASS', patterns: [/gate pass/i, /\bnoc\b/i, /permit/i] },
  { code: 'CRANE_LIFT', patterns: [/crane/i, /forklift/i, /lifting/i] },
  { code: 'DELIVERY_OFFLOAD', patterns: [/delivery/i, /offloading/i, /unloading/i] },
  { code: 'VESSEL_BERTH', patterns: [/vessel/i, /berth/i, /tide/i, /anchorage/i] },
  { code: 'DOC_CUSTOMS', patterns: [/\bboe\b/i, /\bdo\b/i, /customs/i, /fanr/i, /msds/i, /tpi/i] },
  { code: 'WAREHOUSE', patterns: [/\bwh\b/i, /warehouse/i, /yard/i, /storage/i] },
  { code: 'DAMAGE_RETURN', patterns: [/damage/i, /osdr/i, /return/i] },
];

export function classifyMessage(bodyRaw: string): ParsedMessage {
  const text = normalize(bodyRaw);

  const status = detectStatus(text);
  const typeCode = detectType(text);
  const ownerName = extractOwner(text);
  const holdReason = detectHoldReason(text);
  const title = buildTitle(text, typeCode);
  const priorityScore = scorePriority(status, typeCode);
  const dedupeKey = buildDedupeKey(text, typeCode);

  return {
    title,
    status,
    typeCode,
    ownerName,
    holdReason,
    priorityScore,
    dedupeKey,
  };
}

function normalize(input: string) {
  return input.replace(/\s+/g, ' ').trim();
}

function detectStatus(text: string): Status {
  if (DONE_PATTERNS.some((r) => r.test(text))) return 'done';
  if (HOLD_PATTERNS.some((r) => r.test(text))) return 'hold';
  if (PROGRESS_PATTERNS.some((r) => r.test(text))) return 'in_progress';
  if (extractOwner(text)) return 'assigned';
  return 'new';
}

function detectType(text: string): string {
  for (const rule of TYPE_RULES) {
    if (rule.patterns.some((r) => r.test(text))) return rule.code;
  }
  return 'OTHER';
}

function extractOwner(text: string): string | null {
  const explicit =
    text.match(/\b([A-Z][a-z]+)\s+please\s+/) ||
    text.match(/\b([A-Z][a-z]+)\s+to\s+/) ||
    text.match(/@([A-Za-z0-9._-]+)/);

  return explicit?.[1] ?? null;
}

function detectHoldReason(text: string): string | null {
  if (/waiting boe/i.test(text)) return 'WAIT_BOE';
  if (/waiting do/i.test(text)) return 'WAIT_DO';
  if (/gate pass/i.test(text) && /waiting|pending|hold/i.test(text)) return 'WAIT_GATEPASS';
  if (/crane/i.test(text) && /waiting|pending|hold/i.test(text)) return 'WAIT_CRANE';
  if (/no reply|waiting reply/i.test(text)) return 'WAIT_REPLY';
  if (/tide/i.test(text) && /waiting|pending|hold/i.test(text)) return 'WAIT_TIDE';
  return null;
}

function buildTitle(text: string, typeCode: string): string {
  const clipped = text.slice(0, 80);
  return `[${typeCode}] ${clipped}`;
}

function scorePriority(status: Status, typeCode: string): number {
  let score = 50.0;
  if (status === 'hold') score += 20.0;
  if (typeCode === 'DOC_CUSTOMS' || typeCode === 'VESSEL_BERTH') score += 10.0;
  return Math.min(score, 99.0);
}

function buildDedupeKey(text: string, typeCode: string): string | null {
  const ref =
    text.match(/\b[A-Z]{2,5}[-_]\d{3,}\b/)?.[0] ||
    text.match(/\bJ\d{2,3}-\d+\b/i)?.[0] ||
    null;

  return ref ? `${typeCode}:${ref.toUpperCase()}` : null;
}
```

## 8) `lib/db/upsert-work-item.ts`

```ts
import { supabaseAdmin } from '@/lib/supabase-admin';
import type { NormalizedWahaEvent } from '@/lib/waha/normalize-event';
import type { ParsedMessage } from '@/lib/parser/classify-message';

type WorkItemRow = {
  id: string;
  group_id: string;
  source_message_id: string;
  reply_to_message_id: string | null;
  title: string;
  type_code: string;
  status: string;
  owner_name: string | null;
  hold_reason: string | null;
  dedupe_key: string | null;
  priority_score: number;
  last_message_at: string;
};

export async function upsertWorkItemFromEvent(
  event: NormalizedWahaEvent,
  parsed: ParsedMessage
) {
  const activeStatuses = ['new', 'assigned', 'in_progress', 'hold'];
  const since = new Date(Date.now() - 24 * 60 * 60 * 1000).toISOString();

  const { data: candidates, error: candidateError } = await supabaseAdmin
    .from('work_item')
    .select('*')
    .eq('group_id', event.groupId)
    .eq('type_code', parsed.typeCode)
    .in('status', activeStatuses)
    .gte('last_message_at', since)
    .order('last_message_at', { ascending: false })
    .limit(20);

  if (candidateError) throw candidateError;

  const best = chooseBestCandidate(candidates ?? [], event, parsed);

  if (!best) {
    const { data: inserted, error: insertError } = await supabaseAdmin
      .from('work_item')
      .insert({
        group_id: event.groupId,
        source_message_id: event.messageId,
        reply_to_message_id: event.replyToMessageId,
        title: parsed.title,
        type_code: parsed.typeCode,
        status: parsed.status,
        owner_name: parsed.ownerName,
        hold_reason: parsed.holdReason,
        dedupe_key: parsed.dedupeKey,
        priority_score: parsed.priorityScore,
        last_message_at: event.sentAt ?? new Date().toISOString(),
      })
      .select()
      .single();

    if (insertError) throw insertError;

    await supabaseAdmin.from('work_item_status_history').insert({
      work_item_id: inserted.id,
      from_status: null,
      to_status: parsed.status,
      source_message_id: event.messageId,
    });

    return { mode: 'insert', workItemId: inserted.id };
  }

  const { error: updateError } = await supabaseAdmin
    .from('work_item')
    .update({
      status: parsed.status,
      owner_name: parsed.ownerName ?? best.owner_name,
      hold_reason: parsed.holdReason,
      title: parsed.title,
      dedupe_key: parsed.dedupeKey ?? best.dedupe_key,
      priority_score: Math.max(parsed.priorityScore, best.priority_score),
      last_message_at: event.sentAt ?? new Date().toISOString(),
      updated_at: new Date().toISOString(),
    })
    .eq('id', best.id);

  if (updateError) throw updateError;

  if (best.status !== parsed.status) {
    await supabaseAdmin.from('work_item_status_history').insert({
      work_item_id: best.id,
      from_status: best.status,
      to_status: parsed.status,
      source_message_id: event.messageId,
    });
  }

  return { mode: 'update', workItemId: best.id };
}

function chooseBestCandidate(
  candidates: WorkItemRow[],
  event: NormalizedWahaEvent,
  parsed: ParsedMessage
): WorkItemRow | null {
  let best: WorkItemRow | null = null;
  let bestScore = -1;

  for (const c of candidates) {
    let score = 0;

    if (event.replyToMessageId && c.source_message_id === event.replyToMessageId) score += 100;
    if (event.replyToMessageId && c.reply_to_message_id === event.replyToMessageId) score += 30;
    if (parsed.dedupeKey && c.dedupe_key && parsed.dedupeKey === c.dedupe_key) score += 80;
    if (parsed.ownerName && c.owner_name && parsed.ownerName === c.owner_name) score += 10;
    if (c.status === 'hold') score += 5;

    if (score > bestScore) {
      best = c;
      bestScore = score;
    }
  }

  return bestScore >= 30 ? best : null;
}
```

## 9) `app/api/webhooks/waha/route.ts`

```ts
import { NextResponse } from 'next/server';
import { supabaseAdmin } from '@/lib/supabase-admin';
import { verifyWahaSignature } from '@/lib/waha/verify-signature';
import { normalizeWahaEvent } from '@/lib/waha/normalize-event';
import { classifyMessage } from '@/lib/parser/classify-message';
import { upsertWorkItemFromEvent } from '@/lib/db/upsert-work-item';

export const runtime = 'nodejs';
export const maxDuration = 15;

export async function POST(request: Request) {
  const rawBody = await request.text();

  const signatureHeaderName =
    process.env.WAHA_SIGNATURE_HEADER || 'x-waha-signature';

  const signature = request.headers.get(signatureHeaderName);

  const verified = verifyWahaSignature(rawBody, signature);
  if (!verified) {
    return NextResponse.json({ ok: false, error: 'invalid signature' }, { status: 403 });
  }

  let payload: any;
  try {
    payload = JSON.parse(rawBody);
  } catch {
    return NextResponse.json({ ok: false, error: 'invalid json' }, { status: 400 });
  }

  const event = normalizeWahaEvent(payload);
  if (!event) {
    return NextResponse.json({ ok: true, ignored: true, reason: 'not_group_message_or_missing_fields' });
  }

  // 1) raw 저장
  const { error: rawError } = await supabaseAdmin
    .schema('ops_private')
    .from('wa_raw_event')
    .upsert(
      {
        message_id: event.messageId,
        event_type: event.eventType,
        group_id: event.groupId,
        participant_id: event.participantId,
        reply_to_message_id: event.replyToMessageId,
        body_raw: event.bodyRaw,
        payload_json: event.payload,
        sent_at: event.sentAt,
      },
      { onConflict: 'message_id' }
    );

  if (rawError) {
    return NextResponse.json({ ok: false, error: rawError.message }, { status: 500 });
  }

  // 2) 분류
  const parsed = classifyMessage(event.bodyRaw);

  // 3) 카드 upsert
  const result = await upsertWorkItemFromEvent(event, parsed);

  return NextResponse.json({
    ok: true,
    messageId: event.messageId,
    workItemId: result.workItemId,
    mode: result.mode,
    parsed,
  });
}
```

## 12) 결론

1. `raw 먼저 저장`
2. `분류는 앱 레이어`
